<script>
    let selectedP = "";
    let selectedPrice = "";

    function openOrderModal(name, price) {
        selectedP = name;
        selectedPrice = price;
        document.getElementById('pName').innerText = "المنتج: " + name + " - " + price;
        document.getElementById('orderModal').style.display = 'flex';
    }

    function closeModal() {
        document.getElementById('orderModal').style.display = 'none';
    }

    function validateAndSend() {
        const name = document.getElementById('cName').value;
        const phone = document.getElementById('cPhone').value;
        const address = document.getElementById('cAddress').value;

        if(!name || !phone || !address) {
            alert("فضلاً، أكمل بياناتك لنتمكن من توصيل الطلب");
            return;
        }

        const myNumber = "201120106990";
        const text = `*طلب جديد من الموقع* 🛍️%0A%0A` +
                     `*المنتج:* ${selectedP}%0A` +
                     `*السعر:* ${selectedPrice}%0A` +
                     `--------------------------%0A` +
                     `*بيانات العميل:*%0A` +
                     `*الاسم:* ${name}%0A` +
                     `*الموبايل:* ${phone}%0A` +
                     `*العنوان:* ${address}%0A%0A` +
                     `شكراً ho&t!`;

        window.open(`https://wa.me/${myNumber}?text=${text}`, '_blank');
        closeModal();
    }

    // إغلاق المودال عند الضغط خارج المحتوى
    window.onclick = function(event) {
        if (event.target == document.getElementById('orderModal')) closeModal();
    }
</script>
